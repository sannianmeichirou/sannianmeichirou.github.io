函数的参数 ：形参 实参 不定参

function a( ks , ting ){ // 形参
alert( ks + ting );
};
a( '2' ); //实参

function a()
{
//arguments; //不定参 类数组 实参的集合 之力指的是a的集合
//alert( ks + ting );
//alert( arguments[3] );

    var sum = 0;

    for ( var i=0;i<arguments.length;i++ )
    {
        sum += arguments[i];
    }

    alert( sum );
};
a( 2,5,'5',4,6 , 3 ); // 75463

函数作用域与定义

函数作用域与定义   注！：子域可以访问父育的定义和参数  但是父域不能访问子域
（参数是参数不是赋值不是执行 定义仅仅只有定义 参数只有形参 实参 不定参）
    var a = xxxxxxx  用来定义a与赋值
    function a(){xxx} 定义a函数
    var a = function(){xxx}定义a  把函数赋值给a
如果同时出现 var a  跟function a()  即同名定义 以function 为准
如果同时出现 多个 function 则以最后一个function 为准定义 （因为后面的同名function会替换掉前面的）
定义完成之后  便是赋值 即var a = xxxx  这时候 var a = xxx没有定义的意思 只有赋值 赋值function便执行function  
只做定义的function a()没有意义 详情案例见js 07课练习

js里 获取css样式

ie： xx.currentStyle （其他浏览器不支持）    例：  box.currentStyle.width
非ie getComputedStyle(xx) (ie不支持)       例：  getComputedStyle(box).width
注！ css中 background font 等为复合属性  单独值单独获取 例如背景颜色 xxx.backgroundColor
用if三木 让浏览器 js获取样式的兼容方法：
alert(   getStyle(oBox , 'width')   );
function getStyle( obj , attr ){
return obj.currentStyle?obj.currentStyle[attr]:getComputedStyle(obj)[attr]; // if三木运算
注！：if （对象） ture  if（undefined） false

js里 定时器

执行一次： setTimeout(函数/代码串，时间); 相当于延迟执行
重复执行： setInterval(函数/代码串，时间); 
停止定时器 clearTimeout() / clearInterval()
例：    
    var a = setInterval(fn,13);
    var num = 0;
            function fn(){
                console.log('帅！');
                num ++;
                if ( num == 300 )
                {
                    clearInterval( a );
                }
                console.log( num );
            }

for in

json 中 如果要得到json集合中所有的属性 就用for in 语句
例： var a = {
        width:100px;
        height:100px;
        };
    for (var key in a)
    {
        alert(key); // 弹出 width 再弹出 height  即得到a 的属性
        alert（a[key]） // a[key] 即属性的值    
    }

js里设置点击事件的时间间隔（避免点击过快混乱）：

例间隔300毫秒：
      var nowTime = 0;
      xx.onclick = function(){
          if( new Date() - nowTime > 300) //此次时间减去上次时间 即时间间隔//
         {
           nowTime = new Date();        //将此次时间赋值给nowTime 下次点击事件触发 nowTime为上次时间 
            .........
          }  
        }；

js设置开关

例：    var onOff = true;
    for ( var i=0;i<arrClass.length;i++ )
    {
    if ( arrClass[i] == cName )    //当判断 事情A有多种情况 只可能存在（一种或者一种都没有）情况不执行时候
        {                       用开关  不用if else 因为不知道存不存在 
          onOff = false;          此例如果用else(arrClass[i] !== cName)  则在i=0是就执行了else 可能i=1
        }              时候就不行 所以要把i全部完成之后再做判断  else就不合适了 就要用开关
    };
    if ( onOff )obj.className += ' '+cName;
